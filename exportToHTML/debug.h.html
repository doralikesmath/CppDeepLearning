44   NeuralNetwork::Sequential::Sequential(std::vector<int> params) {
45       this->number_of_layers = params.size();
46       for (int i = 0; i < params.size() - 1; i++){
47           this->layers.push_back(Layer(params[i], params[i+1], 0, 0.5));
48       }
49       layers.push_back(Layer(params[params.size() - 1], 1, 0, 0.5));
50       this->compiled = true;
51   }

58   void NeuralNetwork::Sequential::compile(int number_of_epochs, std::string loss) {
59       this->loss = loss;
60       this->number_of_epochs = number_of_epochs;
61       this->number_of_layers = this->params.size();
62       for (int i = 0; i < params.size() - 1; i++){
63           this->layers.push_back(Layer(this->params[i], this->params[i+1], 0, 0.5, this->activations[i]));
64       }
65       layers.push_back(Layer(this->params[this->number_of_layers - 1], 1, 0, 0.5,
66               this->activations[this->number_of_layers-1]));
67       this->compiled = true;
68   }

70   void NeuralNetwork::Sequential::forward_pass() {
71       for (int i = 1; i < this->number_of_layers; i++){
72           this->layers[i].pre_activate = this->layers[i-1].weights.transpose() * this->layers[i-1].post_activate;
73           this->layers[i].activate();
74       }
75   }

77   void NeuralNetwork::Sequential::update_weights(double predicted, double expected) {
78       double temp = predicted - expected;
79       for (int i = 0; i < this->number_of_layers; i++){
80           for (int j = 0; j < this->layers[i].weights.shape[0]; j++){
81               for (int k = 0; k < this -> layers[i].weights.shape[1]; k++){
82                   this->layers[i].weights(j, k) -= this->layers[i].weight_derivatives(j, k) * temp;
83               }
84           }
85       }
86   }

88   void NeuralNetwork::Sequential::back_propagate(double predicted, double expected) {
89       double temp;
90       int n = this->number_of_layers - 1;
91       if (this->loss == "squared_error"){
92           this->layers[n].value_derivatives[0] = (predicted - expected) *
93                   activations::activation_derivative(this->layers[n].post_activate[0],
94                           this->layers[n].activation);
95           std::cout << "dL/da_0 = " << this->layers[n].value_derivatives[0] << std::endl;
96       }
97       for (int i = n - 1; i >= 0; i--){
98           for (int j = 0; j < this->layers[i].value_derivatives.size(); j++){
99               temp = 0;
100              for (int k = 0; k < this->layers[i+1].value_derivatives.size(); k++){
101                  temp += this->layers[i].weights(j, k) * this->layers[i+1].value_derivatives[k];
102              }
103              this->layers[i].value_derivatives[j] = temp *
104                      activations::activation_derivative(this->layers[i].pre_activate[j], this->layers[i].activation);
105          }
106      }
107  
108      for (int i = 0; i < n; i++){
109          for (int j = 0; j < this->layers[i].weights.shape[0]; j++){
110              for (int k = 0; k < this->layers[i].weights.shape[1]; k++){
111                  this->layers[i].weight_derivatives(j, k) = this->layers[i+1].value_derivatives[k] *
112                          this->layers[i].post_activate[j];
113              }
114          }
115      }
116  
117      this->update_weights(predicted, expected);
118  }

121  void NeuralNetwork::Sequential::fit(std::vector<std::vector<Type>> X, std::vector<Type> y) {
122      for (int i = 0; i < this->number_of_epochs; i++){
123          for (int j = 0; j< X.size(); j++){
124              this->layers[0].pre_activate = X[j];
125              this->forward_pass();
126              double predicted = this->layers[this->number_of_layers - 1].post_activate[0];
127              this->back_propagate(predicted, y[j]);
128  
129          }
130      }
