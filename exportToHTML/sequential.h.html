66   NeuralNetwork::Sequential::Sequential(std::vector<int> params) {
67       this->number_of_layers = params.size();
68       for (int i = 0; i < params.size() - 1; i++){
69           this->layers.push_back(Layer(params[i], params[i+1], 0, 0.5));
70       }
71       layers.push_back(Layer(params[params.size() - 1], 1, 0, 0.5));
72       this->compiled = true;
73   }

75   void NeuralNetwork::Sequential::add_layer(int number_of_neuron, std::string activation) {
76       this->params.push_back(number_of_neuron);
77       this->activations.push_back(activation);
78   }

80   void NeuralNetwork::Sequential::compile(int number_of_epochs, std::string loss) {
81       this->loss = loss;
82       this->number_of_epochs = number_of_epochs;
83       this->number_of_layers = this->params.size();
84       for (int i = 0; i < params.size() - 1; i++){
85           this->layers.push_back(Layer(this->params[i], this->params[i+1], 0, 0.5, this->activations[i]));
86       }
87       layers.push_back(Layer(this->params[this->number_of_layers - 1], 1, 0, 0.5,
88               this->activations[this->number_of_layers-1]));
89       this->compiled = true;
90   }

92   void NeuralNetwork::Sequential::forward_pass() {
93       for (int i = 1; i < this->number_of_layers; i++){
94           this->layers[i].pre_activate = this->layers[i-1].weights.transpose() * this->layers[i-1].post_activate;
95           this->layers[i].activate();
96       }
97   }

99   void NeuralNetwork::Sequential::update_weights(double predicted, double expected) {
100      double temp = predicted - expected;
101      for (int i = 0; i < this->number_of_layers; i++){
102          for (int j = 0; j < this->layers[i].weights.shape[0]; j++){
103              for (int k = 0; k < this -> layers[i].weights.shape[1]; k++){
104                  this->layers[i].weights(j, k) -= this->layers[i].weight_derivatives(j, k) * temp;
105              }
106          }
107      }
108  }

110  void NeuralNetwork::Sequential::back_propagate(double predicted, double expected) {
111      double temp;
112      int n = this->number_of_layers - 1;
113      if (this->loss == "squared_error"){
114          this->layers[n].value_derivatives[0] = (predicted - expected) *
115                  activations::activation_derivative(this->layers[n].post_activate[0],
116                          this->layers[n].activation);
117      }
118      for (int i = n - 1; i >= 0; i--){
119          for (int j = 0; j < this->layers[i].value_derivatives.size(); j++){
120              temp = 0;
121              // the dot product of 2 vectors, really
122              for (int k = 0; k < this->layers[i+1].value_derivatives.size(); k++){
123                  temp += this->layers[i].weights(j, k) * this->layers[i+1].value_derivatives[k];
124              }
125              this->layers[i].value_derivatives[j] = temp *
126                      activations::activation_derivative(this->layers[i].pre_activate[j], this->layers[i].activation);
127          }
128      }
129  
130      for (int i = 0; i < n; i++){
131          for (int j = 0; j < this->layers[i].weights.shape[0]; j++){
132              for (int k = 0; k < this->layers[i].weights.shape[1]; k++){
133                  this->layers[i].weight_derivatives(j, k) = this->layers[i+1].value_derivatives[k] *
134                          this->layers[i].post_activate[j];
135              }
136          }
137      }
138  
139      this->update_weights(predicted, expected);
140  }

142  template <typename Type>
143  void NeuralNetwork::Sequential::fit(std::vector<std::vector<Type>> X, std::vector<Type> y) {
144      for (int i = 0; i < this->number_of_epochs; i++){
145          std::cout << "============================================" << std::endl;
146          std::cout << "Epochs " << i << std::endl;
147          for (int j = 0; j< X.size(); j++){
148              // std::cout << "============================================" << std::endl;
149              this->layers[0].pre_activate = X[j];
150              this->forward_pass();
151              double predicted = this->layers[this->number_of_layers - 1].post_activate[0];
152              std::cout << X[j][0] << " XOR " << X[j][1] << " | predicted : " << predicted << " - expected : " << y[j] << std::endl;
153              this->back_propagate(predicted, y[j]);
154          }
155      }
156  }

158  template <typename Type>
159  std::vector<double> NeuralNetwork::Sequential::predict(std::vector<Type> input) {
160      this->layers[0].pre_activate = input;
161      this->forward_pass();
162      return this->layers[this->number_of_layers - 1].post_activate;
163  }

183  void NeuralNetwork::Sequential::summarize() {
184      if (this->compiled) {
185          std::cout << "================ Summarize ================" << std::endl;
186          std::cout << "Number of layers: " << this->number_of_layers << std::endl;
187          std::cout << "============================================" << std::endl;
188          long param = 0;
189          for (int i = 0; i < this->number_of_layers; i++) {
190              std::cout << "Layer " << i << " - Size: " << this->layers[i].weights.shape[0] << " x "
191                        << this->layers[i].weights.shape[1]
192                        << " - Activation: " << this->layers[i].activation << std::endl;
193              param += this->layers[i].weights.shape[0] * this->layers[i].weights.shape[1];
194          }
195          param -= this->layers[this->number_of_layers-1].weights.shape[0] * this->layers[this->number_of_layers-1].weights.shape[1];
196          std::cout << "============================================" << std::endl;
197          std::cout << "Total number of parameters: " << param << std::endl;
198      } else{
199          std::cout << "The model needs to be compiled first." << std::endl;
200      }
201  }
